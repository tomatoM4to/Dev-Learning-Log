async, await

Promise를 좀더 간결하고, 간편하고 동시적으로 실행하는것처럼 보이게 해주는 API

BUT: 모든것을 async로 할필욘 없다, 각각의 장단점에 따라 사용하면 됩니다.

생성

async를 붙이면 해당 함수는 항상 프라미스를 반환합니다. 프라미스가 아닌 값을 반환하더라도 이행 상태의 프라미스(resolved promise)로 값을 감싸 이행된 프라미스가 반환되도록 합니다.

function fetchUser1() {
    return new Promise((resolve, reject) => resolve(`fetchUser1`))
}
 
async function fetchUser2() {
    return `fetchUser2`;
}
 
fetchUser1().then(console.log); // fetchUser1
fetchUser2().then(console.log); // fetchUser2

사용하기

async/await 구문을 사용하면 비동기 코드를 동기식 방식으로 작성할 수 있습니다.

async 함수 내에서 await 키워드를 사용하면 JavaScript는 그 Promise가 settle될 때까지(즉, 해당 작업이 완료되어 결과가 준비될 때까지) 기다리는 것처럼 보입니다.

그러나 실제로는, await 키워드가 앞에 붙은 Promise가 resolve될 때까지 JavaScript는 async 함수의 실행을 "일시 중단"하고, 다른 작업을 수행하러 이벤트 루프로 돌아갑니다. 따라서 async/await 구문은 JavaScript의 비동기성을 유지하면서도 동기적으로 보이는 코드를 작성할 수 있게 해줍니다.

function delay(ms) {
    return new Promise((resolve) => setTimeout(resolve, ms));
}
 
 
async function getApple() {
    await delay(1000);
    return '🍎';
}
 
 
async function getBanana() {
    await delay(1000);
    return '🍌';
}
 
 
// 2초가 걸린다.
function getFruits1() {
    return getApple().then((apple) => {
        return getBanana().then(banana => `${apple}, ${banana}`);
    })
}
 
// 사과를 받는데 1초, 바나나를 받는데 1초 총 2초가 걸린다.
async function getFruits2() {
    const a = await getApple();
    const b = await getBanana();
    return `${a}, ${b}`;
}
 
getFruits1().then(console.log); // 🍎, 🍌
getFruits2().then(console.log); // 🍎, 🍌

호기심 테스트

await는 Promise에 대해 작동하므로, Promise를 반환하지 않는 동기 함수에 대해서는 await가 특별히 무언가를 하지 않습니다. 다시 말해, 동기 함수의 경우 await 키워드를 사용해도 해당 함수의 실행을 "일시 중단"하지 않고, 코드는 여전히 동기적으로 실행됩니다.

function syncFunction() {
    let cnt = 0;
    for(let i = 0; i < 1e9; i++) {
        cnt++;
    }
    return cnt;
}
 
async function awaitTest() {
    console.log('Before');
    const result = await syncFunction(); // This is a synchronous function
    console.log('After');
}
 
 
const btn = document.querySelector(`.btn`);
 
btn.addEventListener('click', () => {
    console.clear();
    awaitTest();
    console.log('Outside');
